# jade #

 1、# 安装依赖 #

    //安装支持pug依赖
    npm install pug pug-loader pug-filters --save-dev
    //安装支持jade依赖
    npm install jade jade-loader --save-dev

 2、# webpack.base.config.js #

    {
      test: /\.jade$/,
      loader: "jade"
    },
    {
      test: /\.pug$/,
      loader: 'pug'
    },
